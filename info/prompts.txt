import { useEffect, useState } from "react";
import gsap from "gsap";
import UserCard from "../components/UserCard";
import { useStore } from "../stores/useStore";
import { SplitText } from "gsap/all";
gsap.registerPlugin(SplitText);
const Home = () => {
  const usersArdaTorre = useStore((state) => state.usersArdaTorre);
  const searched = useStore((state) => state.searched);
  const total = usersArdaTorre.length;
  const [split, setSplit] = useState<SplitText | null>(null);
  useEffect(() => {
    if (!searched) {
      const el = document.querySelector(".text");
      if (!el) return;
      const split = new SplitText(el, { type: "words" });
      setSplit(split);
      gsap.from(split.words, {
        y: 5,
        duration: 1,
        stagger: 0.1,
        autoAlpha: 0,
        filter: "blur(10px)",
      });
    }
    return () => {
      try {
        split?.revert();
      } catch (e) {
        console.warn("SplitText revert error:", e);
      }
    };
  }, []);

  return (
    <div className="max-w-[1080px] h-full mx-auto py-6 flex flex-col gap-4 items-center mt-40">
      {!searched ? (
        <div className="w-full flex flex-col gap-4 items-center justify-center mx-auto">
          <p className="text text-4xl w-[65ch]">
            El mundo laboral ahora es global, y en{" "}
            <span className="text-brand font-semibold">Torre</span>,
            <br /> lo hacemos personal. Tu network profesional empieza aqu√≠.
            <br />
            <br />
            Escribe un nombre y re-conecta.
          </p>
        </div>
      ) : (
        <div className="flex flex-col gap-4">
          {usersArdaTorre.length > 0 ? (
            <>
              <p>Coincidencias encontradas en Torre: {total} profesionales.</p>
              {usersArdaTorre.map((user) => {
                return <UserCard key={user.ardaId} user={user} />;
              })}
            </>
          ) : (
            "No encontramos resultados por ahora. ¬øProbamos con otra b√∫squeda?"
          )}
        </div>
      )}
    </div>
  );
};

export default Home;
